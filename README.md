## python 创建虚拟环境流程
1.python -m venv .venv\
2.Set-ExecutionPolicy -ExecutionPolicy RemoteSigned -Scope Process\
3. .venv\scripts\activate

## 安装sdk包
pip install dhicn-domain-paas-sdk-python
